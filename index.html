while True:
    user_num1 = int(input("Введите ваше первое число: "))
    user_num2 = int(input("Введите ваше второе число: "))
    operator = input("Введите оператора: ")

    if operator == '+':
   
        print('number 1 + number 2 = ', user_num1 + user_num2)
    if operator == '-':
    
        print('number 1 - number 2 = ', user_num1 - user_num2)
    if operator == '*':
    
        print('number 1 * number 2 = ', user_num1 * user_num2)
    if operator == '/':
    
        print('number 1 / number 2 = ', user_num1 / user_num2)
    if operator == '//':
    
        print('number 1 // number 2 = ', user_num1 // user_num2)
    if operator == '**':
    
        print('number 1 ** number 2 = ', user_num1 ** user_num2)
    if operator == '%':
  
        print('number 1 % number 2 = ', user_num1 % user_num2)
